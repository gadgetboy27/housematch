async function getCachedOrGenerate<T>(
  cacheKey: string,
  ttl: number,
  generator: () => Promise<T>
): Promise<T> {
  // Try cache first
  const cached = await redis.get(cacheKey);
  if (cached) {
    console.log('✅ Cache hit:', cacheKey);
    return JSON.parse(cached);
  }
  
  // Generate if not cached
  console.log('💰 AI call:', cacheKey);
  const result = await generator();
  
  // Cache for next time
  await redis.setex(cacheKey, ttl, JSON.stringify(result));
  
  return result;
}

typescript// 80% of recommendations don't need AI
function getSimpleRecommendations(userId: number) {
  const userLikes = getLikedProperties(userId);
  const userDislikes = getDislikedProperties(userId);
  
  // Simple scoring
  const avgBedrooms = average(userLikes.map(p => p.bedrooms));
  const avgPrice = average(userLikes.map(p => p.price));
  const preferredSuburbs = mostCommon(userLikes.map(p => p.suburb), 3);
  
  return properties
    .filter(p => !userDislikes.includes(p.id))
    .map(property => ({
      ...property,
      score: calculateScore(property, {
        avgBedrooms,
        avgPrice,
        preferredSuburbs
      })
    }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 50);
}

function calculateScore(property, userPrefs) {
  let score = 100;
  
  // Bedroom match (40 points)
  score += 40 - Math.abs(property.bedrooms - userPrefs.avgBedrooms) * 10;
  
  // Price match (30 points)
  const priceDiff = Math.abs(property.price - userPrefs.avgPrice) / userPrefs.avgPrice;
  score += 30 * (1 - priceDiff);
  
  // Suburb match (30 points)
  if (userPrefs.preferredSuburbs.includes(property.suburb)) {
    score += 30;
  }
  
  return Math.max(0, score);
}

typescript// server/services/ai-optimized.ts

import { redis } from './redis';
import { openai } from './openai';

// Wrapper for all AI calls
export async function smartAICall<T>(
  feature: string,
  userId: number,
  generator: () => Promise<T>,
  options: {
    cacheDuration?: number; // seconds
    requireMinSwipes?: number;
    useEveryN?: number;
  } = {}
): Promise<T> {
  const {
    cacheDuration = 3600,
    requireMinSwipes = 0,
    useEveryN = 1
  } = options;
  
  // Check if user qualifies for AI
  const swipeCount = await getSwipeCount(userId);
  if (swipeCount < requireMinSwipes) {
    throw new Error('USE_SIMPLE_ALGO');
  }
  
  // Check if we should use AI this time
  if (swipeCount % useEveryN !== 0) {
    throw new Error('USE_SIMPLE_ALGO');
  }
  
  // Check cache
  const cacheKey = `ai:${feature}:${userId}:${Math.floor(swipeCount / useEveryN)}`;
  const cached = await redis.get(cacheKey);
  if (cached) {
    console.log('✅ Cache hit:', cacheKey);
    return JSON.parse(cached);
  }
  
  // Call AI
  console.log('💰 AI call:', cacheKey);
  const startTime = Date.now();
  const result = await generator();
  const duration = Date.now() - startTime;
  
  // Log usage
  await logAIUsage(feature, userId, duration);
  
  // Cache result
  await redis.setex(cacheKey, cacheDuration, JSON.stringify(result));
  
  return result;
}
