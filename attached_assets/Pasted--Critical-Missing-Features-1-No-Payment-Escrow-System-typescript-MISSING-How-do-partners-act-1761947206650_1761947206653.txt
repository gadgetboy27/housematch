async function processServicePayment(orderId: number) {
  const order = await getServiceOrder(orderId);
  const partner = await getPartner(order.partnerId);
  
  // Charge customer
  const paymentIntent = await stripe.paymentIntents.create({
    amount: order.amount * 100, // Convert to cents
    currency: 'nzd',
    customer: order.customerId,
    application_fee_amount: Math.round(order.amount * 0.15 * 100), // 15% commission
    transfer_data: {
      destination: partner.stripeAccountId, // Pay partner directly
    },
  });
  
  // Update order
  await db.update(serviceOrders)
    .set({ 
      paymentIntentId: paymentIntent.id,
      status: 'paid',
      commissionAmount: order.amount * 0.15
    })
    .where(eq(serviceOrders.id, orderId));
}

    .set({ 
      status: 'completed',
      completedAt: new Date()
    })
    .where(eq(serviceOrders.id, orderId));
}

export const serviceSLAs = pgTable('service_slas', {
  serviceType: text('service_type').primaryKey(),
  responseTimeHours: integer('response_time_hours').notNull(),
  completionTimeDays: integer('completion_time_days').notNull(),
  maxActiveOrders: integer('max_active_orders').notNull(),
  cancellationPenalty: decimal('cancellation_penalty').notNull(),
  lateCompletionPenalty: decimal('late_completion_penalty').notNull(),
});

// Track SLA violations
async function checkSLACompliance(orderId: number) {
  const order = await getServiceOrder(orderId);
  const sla = await getSLA(order.serviceType);
  
  const violations = [];
  
  // Check response time
  const responseTime = order.acceptedAt 
    ? (order.acceptedAt.getTime() - order.createdAt.getTime()) / (1000 * 60 * 60)
    : null;
  
  if (!responseTime || responseTime > sla.responseTimeHours) {
    violations.push({
      type: 'late_response',
      penalty: sla.cancellationPenalty * 0.1
    });
  }
  
  // Check completion time
  const completionTime = order.completedAt
    ? (order.completedAt.getTime() - order.acceptedAt.getTime()) / (1000 * 60 * 60 * 24)
    : null;
  
  if (completionTime && completionTime > sla.completionTimeDays) {
    violations.push({
      type: 'late_completion',
      penalty: sla.lateCompletionPenalty
    });
  }
  
  // Apply penalties
  if (violations.length > 0) {
    const totalPenalty = violations.reduce((sum, v) => sum + Number(v.penalty), 0);
    
    await db.update(serviceOrders)
      .set({ 
        slaViolations: violations,
        penaltyAmount: totalPenalty
      })
      .where(eq(serviceOrders.id, orderId));
    
    // Deduct from partner earnings
    await adjustPartnerEarnings(order.partnerId, -totalPenalty);
  }
}

```

```
Service Order: $699

Total per transaction: $105 + $50 + $20 = $175
Plus ongoing: $199 + $18 + $15 = $232/mo per partner
