## Part 1: Setting Up Your API Key

### Environment Variables (.env file)

**Never** put API keys directly in your code. Use environment variables:

```bash
# .env (root of your project)

# LINZ API Configuration
LINZ_API_KEY=your_actual_api_key_here_abc123xyz

# Your API Base URLs
LINZ_WFS_URL=https://data.linz.govt.nz/services
LINZ_ARCGIS_URL=https://services.arcgis.com/xdsHIIxuCWByZiCB/arcgis/rest/services/LINZ_NZ_Property_Titles/FeatureServer/0

# Payment Configuration (we'll set these up next)
STRIPE_SECRET_KEY=sk_test_...
STRIPE_PUBLISHABLE_KEY=pk_test_...
STRIPE_WEBHOOK_SECRET=whsec_...

# Your App Configuration
NEXT_PUBLIC_APP_URL=http://localhost:3000
NODE_ENV=development
```

### Access API Key in Your Code

**Usage in your API routes:**

```javascript
// app/api/property/title/route.js
import { LINZ_CONFIG } from '@/lib/linz-config';
import axios from 'axios';

export async function GET(request) {
  const { searchParams } = new URL(request.url);
  const titleNumber = searchParams.get('titleNumber');
  
  const url = `${LINZ_CONFIG.wfsBaseUrl};key=${LINZ_CONFIG.apiKey}/wfs`;
  
  const params = {
    service: 'WFS',
    version: '2.0.0',
    request: 'GetFeature',
    typeNames: 'layer-50804',
    outputFormat: 'json',
    cql_filter: `title_no='${titleNumber}'`
  };
  
  const response = await axios.get(url, { params });
  
  return Response.json(response.data);
}
```

### Security Best Practices

```javascript
// lib/linz-api.js - Create a secure wrapper

import { LINZ_CONFIG } from './linz-config';
import NodeCache from 'node-cache';

const cache = new NodeCache({ stdTTL: 86400 }); // 24 hour cache

export class LINZApi {
  constructor() {
    if (!LINZ_CONFIG.apiKey) {
      throw new Error('LINZ API key not configured');
    }
  }
  
  async getTitleByNumber(titleNumber) {
    // Check cache first
    const cacheKey = `title:${titleNumber}`;
    const cached = cache.get(cacheKey);
    if (cached) return cached;
    
    // Fetch from LINZ
    const url = `${LINZ_CONFIG.wfsBaseUrl};key=${LINZ_CONFIG.apiKey}/wfs`;
    
    const params = {
      service: 'WFS',
      version: '2.0.0',
      request: 'GetFeature',
      typeNames: 'layer-50804',
      outputFormat: 'json',
      cql_filter: `title_no='${titleNumber}'`
    };
    
    try {
      const response = await axios.get(url, { params });
      
      if (response.data.features && response.data.features.length > 0) {
        const titleData = this.parseTitle(response.data.features[0]);
        cache.set(cacheKey, titleData);
        return titleData;
      }
      
      return null;
    } catch (error) {
      console.error('Error fetching title:', error);
      throw new Error('Failed to fetch title data from LINZ');
    }
  }
  
  parseTitle(feature) {
    return {
      titleNumber: feature.properties.title_no,
      landDistrict: feature.properties.land_district,
      type: feature.properties.type,
      status: feature.properties.status,
      issueDate: feature.properties.issue_date,
      legalDescription: feature.properties.legal_desc,
      area: feature.properties.area,
      geometry: feature.geometry
    };
  }
}

// Export singleton instance
export const linzApi = new LINZApi();
```

### Backend Payment API

```javascript
// app/api/checkout/title-report/route.js

import Stripe from 'stripe';
import { linzApi } from '@/lib/linz-api';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

export async function POST(request) {
  try {
    const { propertyId, titleNumber, reportType } = await request.json();
    
    // Define your pricing
    const prices = {
      basic: 1500, // $15.00 in cents
      comprehensive: 2900 // $29.00 in cents
    };
    
    const amount = prices[reportType] || prices.basic;
    
    // Create Stripe checkout session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      line_items: [
        {
          price_data: {
            currency: 'nzd',
            product_data: {
              name: `${reportType === 'basic' ? 'Basic' : 'Comprehensive'} Title Report`,
              description: `Property title report for ${titleNumber}`,
              images: ['https://housematch.co.nz/images/title-report.png']
            },
            unit_amount: amount
          },
          quantity: 1
        }
      ],
      mode: 'payment',
      success_url: `${process.env.NEXT_PUBLIC_APP_URL}/property/${propertyId}/title-report?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.NEXT_PUBLIC_APP_URL}/property/${propertyId}`,
      metadata: {
        propertyId,
        titleNumber,
        reportType
      }
    });
    
    return Response.json({ sessionId: session.id });
    
  } catch (error) {
    console.error('Error creating checkout session:', error);
    return Response.json(
      { error: 'Failed to create checkout session' },
      { status: 500 }
    );
  }
}
```

### Webhook Handler (Generate Report After Payment)

```javascript
// app/api/webhooks/stripe/route.js

import Stripe from 'stripe';
import { headers } from 'next/headers';
import { linzApi } from '@/lib/linz-api';
import { generateTitleReportPDF } from '@/lib/report-generator';
import { sendTitleReportEmail } from '@/lib/email';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);
const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;

export async function POST(request) {
  const body = await request.text();
  const signature = headers().get('stripe-signature');
  
  let event;
  
  try {
    event = stripe.webhooks.constructEvent(body, signature, webhookSecret);
  } catch (err) {
    console.error('Webhook signature verification failed:', err);
    return Response.json({ error: 'Webhook error' }, { status: 400 });
  }
  
  // Handle successful payment
  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;
    
    // Get metadata from session
    const { propertyId, titleNumber, reportType } = session.metadata;
    const customerEmail = session.customer_details.email;
    
    try {
      // 1. Fetch title data from LINZ
      const titleData = await linzApi.getTitleByNumber(titleNumber);
      
      if (!titleData) {
        throw new Error('Title not found');
      }
      
      // 2. Generate PDF report
      const pdfBuffer = await generateTitleReportPDF({
        titleData,
        reportType,
        propertyId
      });
      
      // 3. Store report in database
      await db.titleReport.create({
        data: {
          propertyId: parseInt(propertyId),
          titleNumber,
          reportType,
          customerEmail,
          stripeSessionId: session.id,
          pdfUrl: await uploadPDFToStorage(pdfBuffer, titleNumber),
          createdAt: new Date()
        }
      });
      
      // 4. Send email with PDF attachment
      await sendTitleReportEmail({
        to: customerEmail,
        titleNumber,
        reportType,
        pdfBuffer
      });
      
      console.log(`Title report generated and sent to ${customerEmail}`);
      
    } catch (error) {
      console.error('Error processing title report:', error);
      // Send error notification to admin
    }
  }
  
  return Response.json({ received: true });
}
```

```javascript
// lib/report-generator.js

import PDFDocument from 'pdfkit';
import { format } from 'date-fns';

export async function generateTitleReportPDF({ titleData, reportType, propertyId }) {
  return new Promise((resolve, reject) => {
    const doc = new PDFDocument({ 
      size: 'A4',
      margins: { top: 50, bottom: 50, left: 50, right: 50 }
    });
    
    const chunks = [];
    
    doc.on('data', chunk => chunks.push(chunk));
    doc.on('end', () => resolve(Buffer.concat(chunks)));
    doc.on('error', reject);
    
    // Header
    doc.fontSize(24)
       .font('Helvetica-Bold')
       .text('Property Title Report', { align: 'center' });
    
    doc.moveDown();
    doc.fontSize(12)
       .font('Helvetica')
       .text(`Generated: ${format(new Date(), 'PPpp')}`, { align: 'center' });
    
    doc.moveDown(2);
    
    // Title Information
    doc.fontSize(16)
       .font('Helvetica-Bold')
       .text('Title Information');
    
    doc.moveDown(0.5);
    
    addField(doc, 'Title Number:', titleData.titleNumber);
    addField(doc, 'Land District:', titleData.landDistrict);
    addField(doc, 'Title Type:', titleData.type);
    addField(doc, 'Status:', titleData.status);
    addField(doc, 'Issue Date:', titleData.issueDate ? format(new Date(titleData.issueDate), 'PP') : 'N/A');
    
    if (titleData.area) {
      addField(doc, 'Area:', `${titleData.area.toLocaleString()} m²`);
    }
    
    doc.moveDown(1);
    
    // Legal Description
    doc.fontSize(16)
       .font('Helvetica-Bold')
       .text('Legal Description');
    
    doc.moveDown(0.5);
    doc.fontSize(10)
       .font('Helvetica')
       .text(titleData.legalDescription || 'Not available', {
         align: 'justify'
       });
    
    // Comprehensive report extras
    if (reportType === 'comprehensive') {
      doc.addPage();
      
      doc.fontSize(16)
         .font('Helvetica-Bold')
         .text('Additional Insights');
      
      doc.moveDown(0.5);
      
      // Add more analysis here
      doc.fontSize(10)
         .font('Helvetica')
         .text('This comprehensive report includes additional analysis and insights...');
    }
    
    // Footer
    doc.fontSize(8)
       .fillColor('#666')
       .text(
         'This report is provided by HouseMatch and is based on data from Land Information New Zealand. ' +
         'For official purposes, please order an official title document from LINZ.',
         50,
         doc.page.height - 100,
         { align: 'center', width: doc.page.width - 100 }
       );
    
    doc.end();
  });
}

function addField(doc, label, value) {
  doc.fontSize(10)
     .font('Helvetica-Bold')
     .text(label, { continued: true })
     .font('Helvetica')
     .text(' ' + value);
  doc.moveDown(0.3);
}
```

## Part 7: Database Schema

```sql
-- PostgreSQL schema for title reports

CREATE TABLE title_reports (
  id SERIAL PRIMARY KEY,
  property_id INTEGER REFERENCES properties(id),
  title_number VARCHAR(50) NOT NULL,
  report_type VARCHAR(20) NOT NULL, -- 'basic' or 'comprehensive'
  customer_email VARCHAR(255) NOT NULL,
  stripe_session_id VARCHAR(255) UNIQUE,
  amount_paid INTEGER NOT NULL, -- in cents
  pdf_url TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  
  INDEX idx_property_id (property_id),
  INDEX idx_title_number (title_number),
  INDEX idx_customer_email (customer_email),
  INDEX idx_stripe_session (stripe_session_id)
);

-- Track LINZ API usage (optional)
CREATE TABLE linz_api_logs (
  id SERIAL PRIMARY KEY,
  endpoint VARCHAR(255),
  query_params JSONB,
  response_time_ms INTEGER,
  success BOOLEAN,
  created_at TIMESTAMP DEFAULT NOW(),
  
  INDEX idx_created_at (created_at)
);
```
